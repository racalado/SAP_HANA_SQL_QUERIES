
SELECT "UTD", COUNT(*) AS "NOTA_COUNT"
FROM "CLP123432"."PLANILHAO_NOTA_CI" 
WHERE ("DATA_ENCE" = '1900-01-01' AND "GRUPO" = 'GRUPO_A' AND "UTD" IS NOT NULL AND  UPPER("TARIFA") NOT LIKE  UPPER('%L%') AND  USERCODE in ('CRIA','DESP','RETI','CRRE','REDI') AND NOTA IS NOT NULL)
GROUP BY UTD
ORDER BY UTD ASC
